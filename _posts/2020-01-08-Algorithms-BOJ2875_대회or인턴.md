        String[] input = bufferedReader.readLine().split(" ");
        int N = Integer.parseInt(input[0]);

        int K = Integer.parseInt(input[2]);
        int total = N + M;

        int team = M;
        while (true){
            if (K <=total-3*team && team*2 <= N) break;
            team --;
        }
        System.out.println(team);
    }
}
